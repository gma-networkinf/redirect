<!DOCTYPE html>

<html>

<head>

    <meta charset="UTF-8">

    <title>Redirecting...</title>

    <script>

        (function() {

            try {

                // 1. Проверка на "старый браузер" (Feature Detection)

                // Если браузер не поддерживает стрелочные функции или Promise, 

                // он выкинет ошибку и перейдет в блок catch.

                if (typeof Promise === "undefined") {

                    throw new Error("Old browser");

                }


                var userAgent = navigator.userAgent || navigator.vendor || window.opera;

                var redirectUrl = "https://checkperson.ouptv.ru/id"; // Ссылка по умолчанию


                // 2. Проверка: Телефон или ПК

                // Ищем признаки мобильных ОС в User Agent

                if (/android|iphone|ipad|ipod|blackberry|iemobile|opera mini/i.test(userAgent.toLowerCase())) {

                    redirectUrl = "https://checkperson.ouptv.ru/mob";

                } else {

                    redirectUrl = "https://checkperson.ouptv.ru/desktop";

                }


                window.location.replace(redirectUrl);


            } catch (e) {

                // Если произошла ошибка (старый браузер), отправляем на /id

                window.location.replace("https://checkperson.ouptv.ru/id");

            }

        })();

    </script>

</head>

<body>

    <p>Перенаправление... Если ничего не произошло, <a href="https://checkperson.ouptv.ru/id">нажмите сюда</a>.</p>

</body>

</html>
